2. Programa Python para gerar massa e dados (teste de ambiente):

import csv
import random
from datetime import datetime, timedelta

# Configuração
random.seed(42)
arquivos = {
    'clientes': ['id_cliente', 'cpf', 'nome', 'cidade', 'estado', 'renda', 'segmento', 'data_atualizacao', 'operacao'],
    'contas': ['id_conta', 'id_cliente', 'tipo_conta', 'status_conta', 'data_abertura', 'data_atualizacao', 'operacao'],
    'cartoes': ['id_cartao', 'id_conta', 'tipo_cartao', 'limite', 'status_cartao', 'data_atualizacao', 'operacao'],
}

# Gera clientes
with open('C:/Edvan/DBC/02_SourceData/data/raw/clientes_cdc.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    writer.writerow(arquivos['clientes'])
    for i in range(1, 101):
        row = [
            i,
            f"{random.randint(100,999)}.{random.randint(100,999)}.{random.randint(100,999)}-{random.randint(0,99):02d}",
            f"Cliente_{i}",
            random.choice(['SP', 'RJ', 'MG', 'RS', 'BA']),
            random.choice(['SP', 'RJ', 'MG', 'RS', 'BA']),
            round(random.uniform(2000, 20000), 2),
            random.choice(['PF', 'PJ', 'VIP']),
            datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
            'I'
        ]
        writer.writerow(row)

print('Clientes gerados com sucesso!')
